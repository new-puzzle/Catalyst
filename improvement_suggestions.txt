# backend/ai_providers.py

class AIRouter:
    """Route to different AIs based on mode and task"""
    
    PROVIDERS = {
        'gemini_flash': {
            'cost': 'lowest',
            'speed': 'fastest',
            'use_for': ['daily_journal', 'quick_responses']
        },
        'deepseek': {
            'cost': 'low',
            'speed': 'fast', 
            'use_for': ['architect_mode', 'drafting']
        },
        'claude': {
            'cost': 'high',
            'speed': 'medium',
            'use_for': ['synthesis', 'deep_analysis', 'scribe_mode']
        },
        'gemini_pro': {
            'cost': 'medium',
            'speed': 'medium',
            'use_for': ['balanced_responses']
        }
    }
    
    def select_provider(self, mode, task_type, force=None):
        """
        Auto-select best AI for the task
        Or let user force a specific one
        """
        if force:
            return force
            
        if mode == "AUTO":
            return "gemini_flash"  # Fast for journaling
        elif mode == "ARCHITECT":
            return "deepseek"  # Good at structure
        elif mode == "SCRIBE":
            return "claude"  # Best writing quality
        elif task_type == "synthesis":
            return "claude"  # Best at connecting dots
```

# backend/providers/base.py

class AIProvider:
    """Base class for all AI providers"""
    
    async def chat(self, message: str, mode: str, system_prompt: str):
        raise NotImplementedError
    
    def estimate_cost(self, tokens: int):
        raise NotImplementedError

# backend/providers/gemini.py
class GeminiProvider(AIProvider):
    def __init__(self):
        self.client = genai.GenerativeModel('gemini-2.0-flash-exp')
        self.cost_per_1k_tokens = 0.0001  # Cheaper
    
    async def chat(self, message, mode, system_prompt):
        response = await self.client.generate_content(
            f"{system_prompt}\n\nUser: {message}"
        )
        return response.text

# backend/providers/claude.py  
class ClaudeProvider(AIProvider):
    def __init__(self):
        self.client = anthropic.Anthropic()
        self.cost_per_1k_tokens = 0.003  # More expensive
    
    async def chat(self, message, mode, system_prompt):
        response = await self.client.messages.create(
            model="claude-sonnet-4-20250514",
            system=system_prompt,
            messages=[{"role": "user", "content": message}]
        )
        return response.content[0].text
```
